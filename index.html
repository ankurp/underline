Underline.js

underline.js is like underscore.js but using modern ES7 syntax. Try it ↓

import { map, reduce } from 'underline'; // Map an array with underline console.log('Result of mapping:', [1, 2, 3]::map((e) => e * 10) ); // Sum an array using underline console.log('Result of reducing:', [1, 2, 3]::reduce((a, e) => a + e, 0) ); // Chaining map and reduce console.log('Result of mapping and reducing:', [1, 2, 3]::map((e) => e * 10) ::reduce((a, e) => a + e, 0) );
