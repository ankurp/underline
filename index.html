Underline.js

underline.js is like underscore.js but using modern ES7 syntax. Try it ↓

"use strict"; import { difference, all, keys } from 'underline'; // Differece of arrays console.log('Result of difference:', [12, 1, 2, 3, 10, 11]::difference([10, 11, 12]) ); // check for all elements are less than 10 returns false console.log('Result of all:', [1, 2, 3, 10]::all((e) => e < 10) ); // Get keys from an object console.log('Result of keys:', { a: 1, b: 2 }::keys() ); // Chaining difference and all console.log('Result of difference and all:', [12, 1, 2, 3, 10, 11]::difference([10, 11, 12]) ::all((e) => e < 10) );
